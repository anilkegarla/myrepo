pipeline {
    
	agent any
  


    tools {
        maven  "maven"
    }

    stages {
	
        stage('code clone') {
            steps {
               git branch: 'main', url: 'https://github.com/anilkegarla/myrepo.git' 
            }
			}

       
	    stage('clean') {
            steps {
              sh 'mvn clean package'  
            }
			}
       
        stage('deploy') {
		
            steps {
              sh 'mvn deploy'
		    } 
            }		    		
    }
}
